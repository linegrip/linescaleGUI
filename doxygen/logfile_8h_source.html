163 template <typename T>
164 T parseNumericField(const QString& input, const QString& fieldName, bool& success) {
165 QStringList splitList = input.split("=");
166 success = false;
167 if (splitList.count() != 2) {
168 return -1;
169 }
170
171 if (splitList[0] != fieldName) {
172 return -1;
173 }
174
175 // Type depending code for int and float
176 if constexpr (std::is_same_v<T, int>) {
177 return splitList[1].toInt(&success);
178 } else if constexpr (std::is_same_v<T, float>) {
179 return splitList[1].toFloat(&success);
180 } else {
181 static_assert(!sizeof(T*), "T must be int or float");
182 }
183 }
